# ── ONNX Runtime 后端 ────────────────────────────────────────────────────────
# 三种包互斥，仅能安装其一：
#   onnxruntime-gpu   — NVIDIA GPU（CUDA 12.x）+ CPU 回退；开发/CI 推荐
#   onnxruntime-directml — 任意 Windows GPU（DirectML）+ CPU 回退；仅限 Windows 分发
#   onnxruntime       — 纯 CPU；Linux / macOS
#
# homr 推理优先级（代码层面）：CUDAExecutionProvider → DmlExecutionProvider → CPUExecutionProvider
#   onnxruntime-gpu 分发：NVIDIA 用户走 CUDA，其余走 CPU
#   onnxruntime-directml 分发：AMD/Intel 用户走 DirectML，其余走 CPU
onnxruntime-gpu>=1.20.0
